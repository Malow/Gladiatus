CREATE TABLE Accounts (id INT NOT NULL AUTO_INCREMENT, 
    username VARCHAR(40) NOT NULL UNIQUE,
    PASSWORD VARCHAR(40) NOT NULL, 
    email VARCHAR(40) NOT NULL, 
    session_id VARCHAR(40),
    PRIMARY KEY (ID));

CREATE TABLE Characters (id INT NOT NULL AUTO_INCREMENT, 
    account_id INT,
    NAME VARCHAR(40) NOT NULL UNIQUE,
    image VARCHAR(40) NOT NULL,
    LEVEL INT NOT NULL,
    xp INT NOT NULL,
    STATUS VARCHAR(100),
    currentHealth FLOAT NOT NULL,
    health FLOAT NOT NULL,
    strength FLOAT NOT NULL,
    dexterity FLOAT NOT NULL, 
    intelligence FLOAT NOT NULL, 
    willpower FLOAT NOT NULL, 
    money INT NOT NULL,
    abilities VARCHAR(100) NOT NULL,
    currentlyTraining VARCHAR(40),
    PRIMARY KEY (ID),
    FOREIGN KEY (account_id) REFERENCES Accounts(id));

CREATE TABLE Abilities (id INT NOT NULL AUTO_INCREMENT, 
    NAME VARCHAR(40) NOT NULL,
    description VARCHAR(100), 
    tags VARCHAR(40), 
    tier INT NOT NULL,
    PRIMARY KEY (ID));
